pipeline {
    agent any
    
    tools {
        maven 'Maven'
    }
    
    stages {
        stage('Build') {
            steps {
                sh 'mvn clean compile'
            }
        }
        
        stage('Test') {
            steps {
                sh 'mvn test -Dselenium.remote=http://selenium-hub:4444/wd/hub'
            }
        }
    }

    post {
        always {
            allure results: [[path: 'target/allure-results']]
            archiveArtifacts artifacts: '**/target/allure-results/**', allowEmptyArchive: true
        }
    }
}
